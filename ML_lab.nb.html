clean_data_str = function(date_str) {
  mdy = mdy(date_str)
  ymd = ymd(date_str)
  
  if(is.na(mdy)){
    if(is.na(ymd)){
      return(NA)
    }
    else {return(ymd)}
  }
  else {return(mdy)}
}

fmt_df_amount = function(df) {
  df = df %>% mutate('Profit' = as.numeric(gsub('[$,]', '', Profit)),
                     'Sales' = as.numeric(gsub('[$,]', '', Sales)))
}
